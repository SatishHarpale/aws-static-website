pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/<your-username>/aws-static-website.git'
            }
        }
        stage('Deploy to S3') {
            steps {
                sh '''
                aws s3 sync . s3://my-static-site-satish-001 --delete
                '''
            }
        }
    }
}
